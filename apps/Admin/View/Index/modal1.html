public function index(){
  $table=MODULE_NAME;
  $columns=C("columns.".MODULE_NAME);
  $M= M($table);
  $select=$M->where()->join()->order();
  R('Comm/Table/index',array($columns,$select));
}

public function delete($M,$ids){
  $id=$M->getPk();
  $result=$M->where($id.' in('.$ids.')')->delete();
  if($result){
       $this->ajaxReturn(1,"success",1);
  }
  else{
       $this->ajaxReturn(0,'fail',0);
  }
}
